// Kết nối database
$host = 'localhost';
$dbname = 'thuadmin';
$username = 'root';
$password = 'root';
$port = 3307; // Cổng được thêm vào
vô open file location của xampp trong thư mục htdoc, tìm xem bỏ flie clone vào chưa,
xong chạy theo đường dẫn http://localhost/webgiay
nếu file clone về tên là webgiay thì chạy http://localhost/webgiay
tên là web thì chạy http://localhost/web
